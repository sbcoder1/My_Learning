
1:CREATE TABLE FOR CPU INFO :

		CREATE TABLE cpuinfo (
			id INT AUTO_INCREMENT PRIMARY KEY,
			physical_cores VARCHAR(100) DEFAULT NULL,
			max_freq VARCHAR(100) DEFAULT NULL,
			min_freq VARCHAR(100) DEFAULT NULL,
			cpu_usage VARCHAR(100) DEFAULT NULL,
			cpu_total VARCHAR(100) DEFAULT NULL
		) ENGINE=InnoDB DEFAULT CHARSET=latin1;

2:CREATE TABLE FOR MEMORY INFO :

		CREATE TABLE memoryinfo (
			id INT AUTO_INCREMENT PRIMARY KEY,
			file_type VARCHAR(100) DEFAULT NULL,
			total_memory VARCHAR(100) DEFAULT NULL,
			available VARCHAR(100) DEFAULT NULL,
			used VARCHAR(100) DEFAULT NULL,
			percentage VARCHAR(100) DEFAULT NULL
		) ENGINE=InnoDB DEFAULT CHARSET=latin1;

3:CREATE TABLE FOR NETWORK INFO :

		CREATE TABLE networkinfo (
			id INT AUTO_INCREMENT PRIMARY KEY,
			hostname VARCHAR(100) DEFAULT NULL,
			ip_address VARCHAR(100) DEFAULT NULL,
			byte_send VARCHAR(100) DEFAULT NULL,
			byte_receive VARCHAR(100) DEFAULT NULL
		) ENGINE=InnoDB DEFAULT CHARSET=latin1;


4:SCREATE TABLE FOR  PLATFORM INFO :

		CREATE TABLE platforminfo (
			id INT AUTO_INCREMENT PRIMARY KEY,
			version VARCHAR(100) DEFAULT NULL,
			system VARCHAR(100) DEFAULT NULL,
			sys_release VARCHAR(100) DEFAULT NULL,
			uname VARCHAR(100) DEFAULT NULL,
			n_node VARCHAR(100) DEFAULT NULL,
			machine VARCHAR(100) DEFAULT NULL,
			processor VARCHAR(100) DEFAULT NULL,
			architecture VARCHAR(100) DEFAULT NULL
		) ENGINE=InnoDB DEFAULT CHARSET=latin1;
